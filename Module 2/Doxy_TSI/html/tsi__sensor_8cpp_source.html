45static void initBaseline(TSIElectrode& elec)
46{
47 uint32_t channel0 = elec.getChannel();
48 TSI0->DATA = ((channel0 << TSI_DATA_TSICH_SHIFT) );
49 TSI0->DATA |= TSI_DATA_SWTS_MASK;
50 while(!(TSI0->GENCS & TSI_GENCS_EOSF_MASK));
51 TSI0->GENCS |= TSI_GENCS_EOSF_MASK;
52 elec.setBaseline(TSI0->DATA & TSI_DATA_TSICNT_MASK);
53}

55void TSIAnalogSlider::selfCalibration(void)
56{
57 TSI0->GENCS |= TSI_GENCS_EOSF_MASK; // Clear End of Scan Flag
58 TSI0->GENCS &= ~TSI_GENCS_TSIEN_MASK; // Disable TSI module
59
60 uint32_t trigger_backup;
61 if(TSI0->GENCS & TSI_GENCS_STM_MASK) { // Back-up TSI Trigger mode from Application
62 trigger_backup = 1;
63 } else {
64 trigger_backup = 0;
65 }
66
67 TSI0->GENCS &= ~TSI_GENCS_STM_MASK; // Use SW trigger
68 TSI0->GENCS &= ~TSI_GENCS_TSIIEN_MASK; // Enable TSI interrupts
69 TSI0->GENCS |= TSI_GENCS_TSIEN_MASK; // Enable TSI module
70
71 initBaseline(_elec0);
72 initBaseline(_elec1);
73
74 TSI0->DATA = ((_elec0.getChannel() << TSI_DATA_TSICH_SHIFT));
75
76 TSI0->GENCS &= ~TSI_GENCS_TSIEN_MASK; // Disable TSI module
77 TSI0->GENCS |= TSI_GENCS_TSIIEN_MASK; // Enale TSI interrupt
78 if (trigger_backup) { // Restore trigger mode
79 TSI0->GENCS |= TSI_GENCS_STM_MASK;
80 } else {
81 TSI0->GENCS &= ~TSI_GENCS_STM_MASK;
82 }
83
84 TSI0->GENCS |= TSI_GENCS_TSIEN_MASK; // Enable TSI module
85 TSI0->DATA |= TSI_DATA_SWTS_MASK;
86}

89void TSIAnalogSlider::sliderRead(void ) {
90 if (_scan_in_progress) {
91 _scan_in_progress = 0;
92 uint32_t delta0 = _elec0.getDelta();
93 uint32_t delta1 = _elec1.getDelta();
94
95 if ((delta0 > _elec0.getThreshold()) || (delta1 > _elec1.getThreshold())) {
96 uint32_t perc_pos0 = (delta0 * 100) / (delta0 + delta1);
97 uint32_t perc_pos1 = (delta1 * 100) / (delta0 + delta1);
98 setSliderPercPosition(0, perc_pos0);
99 setSliderPercPosition(1, perc_pos1);
100 uint32_t dist_pos0 = (perc_pos0 * _range) / 100;
101 uint32_t dist_pos1 = (perc_pos0 * _range) / 100;
102 setSliderDisPosition(0, dist_pos0);
103 setSliderDisPosition(1, dist_pos1);
104
105 setAbsolutePosition(((100 - perc_pos0) + perc_pos1) / 2);
106 setAbsoluteDistance(((_range - dist_pos0) + dist_pos1) / 2);
107 } else {
108 setSliderPercPosition(0, 0);
109 setSliderPercPosition(1, 0);
110 setSliderDisPosition(0, 0);
111 setSliderDisPosition(1, 0);
112 setAbsolutePosition(0);
113 setAbsoluteDistance(0);
114 }
115 }

128static void changeElectrode(TSIAnalogSlider *analog_slider)
129{
130 TSIElectrode* elec = analog_slider->getCurrentElectrode();
131 uint32_t signal = (TSI0->DATA & TSI_DATA_TSICNT_MASK);
132 elec->setSignal(signal);
133
134 TSIElectrode *next_elec = analog_slider->getNextElectrode(elec);
135
136 analog_slider->setCurrentElectrode(next_elec);
137 TSI0->DATA = ((next_elec->getChannel() << TSI_DATA_TSICH_SHIFT) );
138 TSI0->DATA |= TSI_DATA_SWTS_MASK;
139}
