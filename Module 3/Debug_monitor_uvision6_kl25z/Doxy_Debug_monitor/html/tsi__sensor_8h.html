#include "mbed.h"
#include "TSISensor.h"
int main(void) {
DigitalOut led(LED_GREEN);
TSIElectrode elec0(9);
TSIElectrode elec1(10);
TSIAnalogSlider tsi(elec0, elec1, 40);
while (true) {
printf("slider percentage: %f%\r\n", tsi.readPercentage());
printf("slider distance: %dmm\r\n", tsi.readDistance());
wait(1);
led = !led;
}
}
