187void chk_UART_msg(void)
188{
189 UCHAR j;
190 while( UART_input() ) // becomes true only when a byte has been received
191 { // skip if no characters pending
192 j = UART_get(); // get next character
193
194 if( j == '\r' ) // on a enter (return) key press
195 { // complete message (all messages end in carriage return)
196 UART_msg_put("->");
197 UART_msg_process();
198 }
199 else
200 {
201 if ((j != 0x02) ) // if not ^B
202 { // if not command, then
203 UART_put(j); // echo the character
204 }
205 else
206 {
207 ;
208 }
209 if( j == '\b' )
210 { // backspace editor
211 if( msg_buf_idx != 0)
212 { // if not 1st character then destructive
213 UART_msg_put(" \b");// backspace
214 msg_buf_idx--;
215 }
216 }
217 else if( msg_buf_idx >= MSG_BUF_SIZE )
218 { // check message length too large
219 UART_msg_put("\r\nToo Long!");
220 msg_buf_idx = 0;
221 }
222 else if ((display_mode == QUIET) && (msg_buf[0] != 0x02) &&
223 (msg_buf[0] != 'D') && (msg_buf[0] != 'N') &&
224 (msg_buf[0] != 'V') &&
225 (msg_buf_idx != 0))
226 { // if first character is bad in Quiet mode
227 msg_buf_idx = 0; // then start over
228 }
229 else { // not complete message, store character
230
231 msg_buf[msg_buf_idx] = j;
232 msg_buf_idx++;
233 if (msg_buf_idx > 2)
234 {
235 UART_msg_process();
236 }
237 }
238 }
239 }
