73void serial(void) // The serial function polls the serial port for
74 // received data or data to transmit
75{
76 // deals with error handling first
77 if ( OERR ) // if an overrun error, clear it and continue.
78 {
79 error_count++;
80 // resets and sets continous receive enable bit
81 UART0->C2 = UART0->C2 & (!UARTLP_C2_RE_MASK);
82 UART0->C2 = UART0->C2 | UARTLP_C2_RE_MASK;
83 }
84
85 if ( FERR){ // if a framing error, read bad byte, clear it and continue.
86 error_count++;
87 RCREG; // This will also clear RCIF if only one byte has been
88 // received since the last int, which is our assumption.
89
90 // resets and sets continous receive enable bit
91 UART0->C2 = UART0->C2 & (!UARTLP_C2_RE_MASK);
92 UART0->C2 = UART0->C2 | UARTLP_C2_RE_MASK;
93 }
94 else // else if no frame error,
95 {
96 if ( RCIF ) // Check if we have received a byte
97 { // Read byte to enable reception of more bytes
98 // For PIC, RCIF automatically cleared when RCREG is read
99 // Also true of Freescale KL25Z
100 *rx_in_ptr++ = RCREG; /* get received character */
101 if( rx_in_ptr >= RX_BUF_SIZE + rx_buf )
102 {
103 rx_in_ptr = rx_buf; /* if at end of buffer, circles rx_in_ptr
104 to top of buffer */
105 }
106
107 }
108 }
109
110 if (TXIF) // Check if transmit buffer empty
111 {
112 if ((tx_in_ptr != tx_out_ptr) && (display_mode != QUIET))
113 {
114 TXREG = *tx_out_ptr++; /* send next char */
115 if( tx_out_ptr >= TX_BUF_SIZE + tx_buf )
116 tx_out_ptr = tx_buf; /* 0 <= tx_out_idx < TX_BUF_SIZE */

122 }

145/*******************************************************************************
146* The function UART_put puts a byte, to the transmit buffer at the location
147* pointed to by tx_in_idx. The pointer is incremented circularly as described
148* previously. If the transmit buffer should wrap around (should be designed
149* not to happen), data will be lost. The serial interrupt must be temporarily
150* disabled since it reads tx_in_idx and this routine updates tx_in_idx which is
151* a 16 bit value.
152*******************************************************************************/
153void UART_put(UCHAR c)
154{
155 *tx_in_ptr++ = c; /* save character to transmit buffer */
156 if( tx_in_ptr >= TX_BUF_SIZE + tx_buf)
157 tx_in_ptr = tx_buf; /* 0 <= tx_in_idx < TX_BUF_SIZE */
158}

160/*******************************************************************************
161* The function UART_get gets the next byte if one is available from the receive
162* buffer at the location pointed to by rx_out_idx. The pointer is circularly
163* incremented and the byte is returned in R7. Should no byte be available the
164* function will wait until one is available. There is no need to disable the
165* serial interrupt which modifies rx_in_idx since the function is looking for a
166* compare only between rx_in_idx & rx_out_idx.
167*******************************************************************************/
168UCHAR UART_get(void)
169{
170 UCHAR c;
171 while( rx_in_ptr == rx_out_ptr ); /* wait for a received character,
172 indicated */
173 // when pointers are different
174 // this could be an infinite loop, but
175 // is not because of UART_input check
176 c = *rx_out_ptr++;
177 if( rx_out_ptr >= RX_BUF_SIZE + rx_buf ) // if at end of buffer
178 {
179 rx_out_ptr = rx_buf; /* 0 <= rx_out_idx < RX_BUF_SIZE */
180 // return byte from beginning of buffer
181 } // next time.
182 return(c);
183}

185/*******************************************************************************
186* The function UART_input returns a 1 if 1 or more receive byte(s) is(are)
187* available and a 0 if the receive buffer rx_buf is empty. There is no need to
188* disable the serial interrupt which modifies rx_in_idx since function is
189* looking for a compare only between rx_in_idx & rx_out_idx.
190*******************************************************************************/
191UCHAR UART_input(void)
192{
193 if( rx_in_ptr == rx_out_ptr )
194 return(0); /* no characters in receive buffer */
195 else
196 return(1); /* 1 or more receive characters ready */
197}

232/*******************************************************************************
233* HEX_TO_ASC Function
234* Function takes a single hex character (0 thru Fh) and converts to ASCII.
235******************************************************************************/
236UCHAR hex_to_asc(UCHAR c)
237{
238 if( c <= 9 )
239 return( c + 0x30 );
