177/*************************************************************************************************************
178 *
179 * Name : static int adc_cal(void)
180 *
181 *
182 *
183 * Description : This function initializes ADC register to calibrate the ADC for stable values
184 *
185 * Inputs: NONE
186 *
187 * Return: Calibration Done or fail response
188 *************************************************************************************************************/
189static int adc_cal(void)
190{
191 uint16_t calib;
192
193 ADC0->CFG1 |= (ADC_CFG1_MODE(3) | // 16 bits mode
194 ADC_CFG1_ADICLK(1)| // Input Bus Clock divided by 2 (48Mhz / 2)
195 ADC_CFG1_ADIV(3)); // Clock divide by 8 (3 MHz)
196
197 ADC0->SC3 |= (ADC_SC3_AVGE_MASK | // Enable HW average
198 ADC_SC3_AVGS(3) | // Set HW average of 32 samples
199 ADC_SC3_CAL_MASK); // Start calibration process
200
201 while (ADC0->SC3 & ADC_SC3_CAL_MASK); // Wait for calibration to end
202
203 if (ADC0->SC3 & ADC_SC3_CALF_MASK) // Check for successful calibration
204 return -1;
205
206 calib = 0;
207 calib += ADC0->CLPS + ADC0->CLP4 + ADC0->CLP3 +
208 ADC0->CLP2 + ADC0->CLP1 + ADC0->CLP0;
209 calib /= 2;
210 calib |= 0x8000; // Set MSB
211 ADC0->PG = calib;
212
213 calib = 0;
214 calib += ADC0->CLMS + ADC0->CLM4 + ADC0->CLM3 +
215 ADC0->CLM2 + ADC0->CLM1 + ADC0->CLM0;
216 calib /= 2;
217 calib |= 0x8000; // Set MSB
218 ADC0->MG = calib;
219
220 return 0;
221}
